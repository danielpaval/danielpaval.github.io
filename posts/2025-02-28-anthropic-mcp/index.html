    Elasticsearch#

    1. Install Python
    2. Install uv: pip install uv
    3. Clone https://github.com/cr7258/elasticsearch-mcp-server
    4. Setup Claude

    uvx Setup#

    {
    "mcpServers": {
        "elasticsearch": {
            "command": "uvx",
            "args": [
                "elasticsearch-mcp-server"
            ],
            "env": {
                "ELASTIC_HOST": "http://localhost:9200",
                "ELASTIC_USERNAME": "elastic",
                "ELASTIC_PASSWORD": "changeme"
            }
        }
    }
}

    uv Setup#

    {
    "mcpServers": {
      "elasticsearch": {
        "command": "uv",
        "args": [
          "--directory",
          "C:\\Repositories\\mcp\\elasticsearch-mcp-server",
          "run",
          "elasticsearch-mcp-server"
        ],
        "env": {
          "ELASTIC_HOST": "http://localhost:9200",
          "ELASTIC_USERNAME": "elastic",
          "ELASTIC_PASSWORD": "changeme"
        }
      }
    }
  }
